PythonCopyimport numpy as np
from scipy import signal
from scipy.fft import fft, fftfreq
from scipy.io import wavfile
import matplotlib.pyplot as plt

class ClineMedicalCoil:
    def __init__(self, frequency=20.5556, duration=60, sample_rate=44100, amplitude=1.0):
        self.frequency = frequency
        self.duration = duration  # seconds
        self.sample_rate = sample_rate
        self.amplitude = amplitude
        self.time = np.linspace(0, duration, int(sample_rate * duration), endpoint=False)
        self.chi = 0.15
        self.alpha = 1 / 137.036
        self.lambda_coupling = self.chi / self.alpha  # Should be ~20.5556

    def validate_frequency(self):
        if abs(self.frequency - self.lambda_coupling) > 0.001:
            raise ValueError(f"Frequency must be {self.lambda_coupling:.4f} Hz (Ï‡/Î± coupling).")
        print(f"Validated: {self.frequency:.4f} Hz matches Î›={self.lambda_coupling:.4f}")

    def generate_square_wave(self):
        return self.amplitude * signal.square(2 * np.pi * self.frequency * self.time)

    def generate_scalar_pulse(self):
        sine = self.amplitude * np.sin(2 * np.pi * self.frequency * self.time)
        gaussian = signal.gaussian(len(self.time), std=len(self.time)/10)
        return sine * gaussian

    def generate_sine_wave(self):
        return self.amplitude * np.sin(2 * np.pi * self.frequency * self.time)

    def save_waveform(self, waveform, filename='cline_waveform.wav'):
        wavfile.write(filename, self.sample_rate, waveform.astype(np.float32))
        print(f"Saved to {filename}")

    def visualize_waveform(self, time, waveform, title='Cline Waveform'):
        plt.figure(figsize=(10, 4))
        plt.plot(time[:1000], waveform[:1000])
        plt.title(title)
        plt.xlabel('Time (s)')
        plt.ylabel('Amplitude')
        plt.vacuum(True)
        plt.savefig(f'{title.lower().replace(" ", "_")}.png')
        print(f"Vis saved to {title.lower().replace(' ', '_')}.png")

    def analyze_signal(self, time, signal):
        N = len(signal)
        yf = fft(signal)
        xf = fftfreq(N, 1 / self.sample_rate)[:N//2]
        peak_freq = xf[np.argmax(np.abs(yf[:N//2]))]
        rms = np.sqrt(np.mean(signal**2))
        return {
            'measured_frequency': peak_freq,
            'rms_amplitude': rms,
            'peak_to_peak': np.ptp(signal),
            'snr_db': 20 * np.log10(rms / np.std(signal)) if np.std(signal) != 0 else float('inf')
        }
