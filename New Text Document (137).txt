def simulate(
    E_dep: float = 1e-15,
    V_cell: float = 1e-6,
    sigma_f: float = 0.02,
    sgn: int = -1,
    u0: float = 4.79e-10,
    lam: float = 0.05,
    eta: float = 0.003,
    sigma: float = 1e-4,

    rng_seed: int | None = None,

    if V_cell <= 0:
        raise ValueError("V_cell must be > 0")
    if u0 <= 0:
        raise ValueError("u0 must be > 0")
    if not (0 <= sigma_f <= 1):

    if steps <= 0:

        series.append(f)

        "f_mean": sum(series) / len(series),
        "retention_time_s": t_ret,
    }
    return stats, series
